class User(context: Context) {

    private val prefs: SharedPreferences = context.getSharedPreferences("user_class", Context.MODE_PRIVATE)

    var pinCode: String?
        get() = prefs.getString("pinCode", null)
        set(value) {
            prefs.edit().putString("pinCode", value).apply()
        }
}
